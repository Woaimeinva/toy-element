<script setup>
import { ref } from 'vue'
const text = ref('组合使用')
const fontFamily = ref('Microsoft YaHei')
const fontSize = ref('18px')
const fontWeight = ref('bold')
const color = ref('rgba(255, 0, 0, 0.2)')
const xGap = ref(80)
const yGap = ref(120)
const rotate = ref(10)
</script>
<template>
    <div class="container">
        <er-watermark 
            :text="text" 
            :fontFamily="fontFamily" 
            :fontSize="fontSize" 
            :fontWeight="fontWeight"
            :color="color"
            :xGap="xGap" 
            :yGap="yGap" 
            :rotate="rotate">
            <template #content>
                <div class="content-panel">
                    <h3>组合使用示例</h3>
                    <div class="controls">
                        <div class="control-item">
                            <span>文字：</span>
                            <input type="text" v-model="text">
                        </div>
                        <div class="control-item">
                            <span>字体：</span>
                            <input type="text" v-model="fontFamily">
                        </div>
                        <div class="control-item">
                            <span>字体大小：</span>
                            <input type="text" v-model="fontSize">
                        </div>
                        <div class="control-item">
                            <span>字体粗细：</span>
                            <input type="text" v-model="fontWeight">
                        </div>
                        <div class="control-item">
                            <span>颜色：</span>
                            <input type="text" v-model="color">
                        </div>
                        <div class="control-item">
                            <span>旋转角度：</span>
                            <input type="number" v-model="rotate">
                        </div>
                        <div class="control-item">
                            <span>X轴间距：</span>
                            <input type="number" v-model="xGap">
                        </div>
                        <div class="control-item">
                            <span>Y轴间距：</span>
                            <input type="number" v-model="yGap">
                        </div>
                    </div>
                    <div class="preview-info">
                        <p>当前设置：</p>
                        <p>字体：{{ fontFamily }}，{{ fontSize }}，{{ fontWeight }}</p>
                        <p>颜色：{{ color }}</p>
                        <p>间距：X轴{{ xGap }}px，Y轴{{ yGap }}px</p>
                        <p>旋转：{{ rotate }}度</p>
                    </div>
                </div>
            </template>
        </er-watermark>
    </div>
</template>
